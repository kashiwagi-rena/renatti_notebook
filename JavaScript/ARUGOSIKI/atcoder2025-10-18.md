function Main(input) {
  const [S, A, B, X] = input.trim().split(" ");
  const ml = Number(S) * Number(A);
  const division = Math.floor(Number(X) / (Number(A) + Number(B)));
  const divi = (Number(A) + Number(B));
  const dev = Number(X) - (divi * division)
  if (dev > Number(A)) {
    const n = dev - Number(A);
    const m = dev - n;
    console.log((ml * division) + ( m * Number(S)));
    return;
  }
  
  console.log((ml * division) + dev * Number(S));
}

Main(require("fs").readFileSync("/dev/stdin", "utf8"));
